  • 做scheme ci的时候，顺便学习了一下bash/sub process/global varible
  • 按照官网的说明写了安装和测试的脚本：
#!/bin/bash

VERSION="9.2"

wget "http://ftp.gnu.org/gnu/mit-scheme/stable.pkg/${VERSION}/mit-scheme-${VERSION}-x86-64.tar.gz"

tar xzf mit-scheme-${VERSION}-x86-64.tar.gz

cd mit-scheme-${VERSION}/src

./configure
make
sudo make install
#!/bin/bash

run_exit=0

for scheme_file in $(find ./* -type f -name '*.scm');
do
   echo "$scheme_file test..."
   mit-scheme < $scheme_file 1>/dev/null 2>&1
   ret_code=$?
   if [ $ret_code != 0 ]; then
        echo ""
        echo "run $scheme_file failed!"
        echo ""
       run_exit=$(($run_exit + 1))
   fi
done
echo $run_exit
exit $run_exit
